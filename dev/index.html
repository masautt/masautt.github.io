Dev environment

How my code is organized on every machine: one predictable tree where folder = GitHub owner. The shape is data (layout.json), reproduced by a two-command bootstrap — so no single machine is "the main one."

The resolver — resolve.mjs

Turns a <owner>/<repo> slug into its on-disk path, applying every placement mode above. Rendered from the live file in masautt-inc/config, so it never drifts.

// Canonical Node resolver. Ports resolve_dest.py's resolve_rel (kept in lockstep by

// bootstrap/conformance/test/resolve.test.mjs). The PS/Python mirrors are slated to

// retire in favor of this (Node-only tooling direction).

function explicitSub(owner, repo, layout) {
  const eg = layout.explicit_groups?.[owner];
  if (!eg) return null;
  for (const [path, repos] of Object.entries(eg.placement || {})) {
    if ((repos || []).includes(repo)) return `${path}/${repo}`;
  }
  return null;                                            // explicit owner, repo unlisted → org root

}

function familySub(owner, repo, layout) {
  const fg = layout.family_grouped?.[owner];
  if (!fg || !repo) return repo;
  const fam = repo.split('-')[0];
  if (!(fg.families || []).includes(fam)) return repo;
  if (repo.endsWith('-template')) return `${fam}/templates/${repo}`;
  if (repo.endsWith('-comp')) return `${fam}/comps/${repo}`;
  if (['finapp', 'minapp', 'pinapp'].includes(fam) && (repo.match(/-/g) || []).length >= 2) return `${fam}/comps/${repo}`;
  return `${fam}/${repo}`;
}

export function resolveRel(slug, layout) {
  const i = slug.indexOf('/');
  const owner = i < 0 ? slug : slug.slice(0, i);
  const repo = i < 0 ? '' : slug.slice(i + 1);
  let sub = explicitSub(owner, repo, layout);
  if (sub === null) sub = familySub(owner, repo, layout);
  const house = layout.also_clone?.[owner]?.promotes_to;
  return house ? `${house}/${owner}/${sub}` : `${owner}/${sub}`;
}
